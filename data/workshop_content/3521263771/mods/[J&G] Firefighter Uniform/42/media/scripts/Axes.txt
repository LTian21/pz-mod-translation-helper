        tags            = AnySurfaceCraft,
        SkillRequired = MetalWelding:3,
        timedAction = Welding_Surface,

        category = Weaponry,
        inputs
        {
	    item 2 [Base.BlowTorch],

            item 1 [Base.SmallSheetMetal],
            item 1 [Base.LeatherStrips] mode:destroy,
